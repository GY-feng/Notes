扩展欧几里得求逆元
```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long int lli
lli p=998244353;
lli mod(lli number){return number%p;}
void exgcd(lli a,lli b,lli& x,lli& y)
{
    if(b==0)
    {
        x=1;y=0;return;
    }
    exgcd(b,a%b,y,x);
    y-=(a/b)*x;
}
lli inv(lli number,lli p)
{
    lli x=1,y=0;
    exgcd(number,p,x,y);
    return (x%p+p)%p;
}
signed main()
 {
    lli k;cin>>k;
    lli k_inv=inv(k,p);
    cout<<((k_inv*k)%p==1);
 }
```
